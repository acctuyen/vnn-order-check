  <h2>Tra cứu thông tin đơn hàng</h2>
  <input type="text" id="orderInput" placeholder="Nhập Original Order SN">
  <button onclick="timKiem()">Kiểm tra</button>
  <div id="loading">Đang lấy dữ liệu...</div>
  <div id="ketQua"></div>

  <script>
    function timKiem() {
      const ma = document.getElementById("orderInput").value.trim();
      const loading = document.getElementById("loading");
      const ketQua = document.getElementById("ketQua");
      if (!ma) return alert("Vui lòng nhập mã đơn hàng.");

      ketQua.innerHTML = "";
      loading.style.display = "block";

      fetch(`https://script.google.com/a/macros/shopee.com/s/AKfycbwr3awGN4tl2b9itdNhCIba-WsTxK34WFyF4LbycQ0ci9StiykOny6p6VGahOT7a9TdfA/exec?order=${ma}`)
        .then(res => res.json())
        .then(data => {
          loading.style.display = "none";
          if (data.result.length === 0) {
            ketQua.innerHTML = "<p>Không tìm thấy dữ liệu cho mã đơn này.</p>";
            return;
          }

          let html = `<h3>Kết quả: ${data.order}</h3><table><tr><th>SKU_ID</th><th>Follow Up</th><th>Return Qty</th></tr>`;
          data.result.forEach(row => {
            html += `<tr><td>${row.SKU_ID}</td><td>${row.FollowUp}</td><td>${row.ReturnQty}</td></tr>`;
          });
          html += "</table>";
          ketQua.innerHTML = html;
        })
        .catch(err => {
          loading.style.display = "none";
          ketQua.innerHTML = `<p style="color:red;">Lỗi khi lấy dữ liệu.</p>`;
        });
    }
  </script>
