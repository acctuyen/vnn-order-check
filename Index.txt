<!DOCTYPE html>
<html>
<head>
  <title>Check Order SN</title>
  <meta charset="UTF-8">
  <script>
    async function searchOrder() {
      const input = document.getElementById("orderInput").value.trim();
      const res = await fetch("data.json");
      const data = await res.json();
      const resultDiv = document.getElementById("result");
      const found = data.find(item => item.OrderSN === input);

      if (found) {
        resultDiv.innerHTML = `
          <p><strong>SKU_ID:</strong> ${found.SKU_ID}</p>
          <p><strong>Follow Up Action:</strong> ${found.Follow}</p>
          <p><strong>Return Qty:</strong> ${found.Qty}</p>
        `;
      } else {
        resultDiv.innerHTML = "<p style='color:red'>Không tìm thấy đơn hàng.</p>";
      }
    }
  </script>
</head>
<body>
  <h1>Tra cứu đơn hàng theo Order SN</h1>
  <input id="orderInput" placeholder="Nhập Order SN" />
  <button onclick="searchOrder()">Tìm</button>
  <div id="result" style="margin-top: 20px;"></div>
</body>
</html>
